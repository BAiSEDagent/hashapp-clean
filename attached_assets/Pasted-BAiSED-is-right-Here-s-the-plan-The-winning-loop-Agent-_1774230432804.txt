const HASHAPP_URL = process.env.HASHAPP_URL || "https://hashapp-prototype.replit.app";
const AGENT_KEY = process.env.AGENT_API_KEY;

// Private context — this is what Venice reasons over privately
// In a real agent this would be live market data, internal docs, etc.
const PRIVATE_CONTEXT = `
Current wallet state: $60 USDC available
Today's spending: $0 of $89 daily cap
Vendor: DataStream Pro
Price: $5/session
Last used: 2 days ago, report quality: high
Internal budget note: market data critical for today's analysis
`;

async function reason(prompt: string): Promise<string> {
  const res = await fetch(`${HASHAPP_URL}/api/gateway/reason`, {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      Authorization: `Bearer ${AGENT_KEY}`,
    },
    body: JSON.stringify({
      messages: [{ role: "user", content: prompt }],
      system: `You are a research agent reasoning over private wallet context. 
               Be concise. Output only: APPROVE <amount> <vendor> <reason> 
               or DENY <reason>.`,
    }),
  });
  const chunks: string[] = [];
  const reader = res.body?.getReader();
  const decoder = new TextDecoder();
  while (reader) {
    const { done, value } = await reader.read();
    if (done) break;
    chunks.push(decoder.decode(value));
  }
  return chunks.join("");
}

async function sendSpendRequest(vendor: string, amount: number, reason: string) {
  await fetch(`${HASHAPP_URL}/api/gateway/message`, {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      Authorization: `Bearer ${AGENT_KEY}`,
    },
    body: JSON.stringify({
      type: "spend_request",
      content: reason,
      metadata: { vendor, amount },
    }),
  });
}

async function pollForApproval(): Promise<boolean> {
  let since = 0;
  for (let i = 0; i < 60; i++) { // poll for 3 minutes
    await new Promise(r => setTimeout(r, 3000));
    const res = await fetch(
      `${HASHAPP_URL}/api/gateway/messages?since=${since}`,
      { headers: { Authorization: `Bearer ${AGENT_KEY}` } }
    );
    const { messages } = await res.json();
    for (const m of messages) {
      if (m.content?.toLowerCase().includes("approved")) return true;
      if (m.content?.toLowerCase().includes("denied")) return false;
      since = m.ts;
    }
  }
  return false;
}

async function main() {
  console.log("Agent: reasoning privately with Venice...");
  
  const decision = await reason(
    `Given this private context, should I request a DataStream Pro session?\n${PRIVATE_CONTEXT}`
  );
  
  console.log("Venice private reasoning:", decision);
  
  if (decision.startsWith("APPROVE")) {
    const parts = decision.split(" ");
    const amount = parseFloat(parts[1]) || 5;
    const vendor = parts[2] || "DataStream Pro";
    const rationale = parts.slice(3).join(" ");
    
    console.log(`Agent: sending spend request — ${vendor} $${amount}`);
    await sendSpendRequest(vendor, amount, rationale);
    
    console.log("Agent: waiting for human approval...");
    const approved = await pollForApproval();
    
    console.log(approved ? "Agent: approved, executing spend" : "Agent: denied");
  } else {
    console.log("Agent: Venice recommended deny — not sending request");
  }
}

main();
